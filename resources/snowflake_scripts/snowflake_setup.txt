---creation of database,schema,role,warehouse-------
create database dbt_project;
create or replace schema dbt_project.raw;
create warehouse transform with warehouse_size = 'Medium';
create role dbt_role;

----grant permissions to the role----
grant usage on database dbt_project to role dbt_role;
grant modify on database dbt_project to role dbt_role;
grant monitor on database dbt_project to role dbt_role;
grant create schema on database dbt_project to role dbt_role;


-----grant role to current_user-----
select current_user();
grant role dbt_role to user PRANJD302;
